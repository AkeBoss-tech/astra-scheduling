	Sys.Application.add_load(InitPage);
	
	function InitPage(){
		details = $find('ctl00_mainContent_courseDetails');
	}
	
	var details;

	function ViewCourseByID(id, termCode, allowAdd, courseTaken) {
		// When viewing details for a course in a requirement (not a Completed Course),
		// Rutgers want to see the latest version of the course only - calling
		// GetCourse will do that. #67702.
		var forceGetByCode = false;
		if (allowAdd == true && courseTaken == false) { // true, false = view requirement course; false, true = completed course.
			forceGetByCode = true;
		}

		if (allowAdd) details.enablePlan();
		else details.disablePlan();

		details.showProgress();
		details.get_ModalForm().show();

		PageMethods.set_path('/Content/Rutgers/Controls/DN/Audit/DegreeAudit.aspx');		
		PageMethods.GetCourse(id, termCode, '0', isContextUser, courseTaken, forceGetByCode, ViewCourseSuccess, deac.pageMethodFailed, allowAdd);
	}

	function ViewCourse(code, termCode, allowAdd, courseTaken) {
		if (allowAdd)	details.enablePlan();
		else details.disablePlan();

		details.showProgress();
		details.get_ModalForm().show();

		PageMethods.set_path('/Content/Rutgers/Controls/DN/Audit/DegreeAudit.aspx');	
		PageMethods.GetCourseEx(code, termCode, '0', isContextUser, courseTaken, ViewCourseSuccess, deac.pageMethodFailed, allowAdd);
	}

	function ViewCourseSuccess(result, allowAdd) {
		details.setCourseDetails(result.Course.ID, result.Course.Code, result.Course.Title, result.Course.Credits, result.Course.FullDescription, result.Course.ExclusivityCoursePlanDetails, result.Course.ExclusivityTranscriptDetails, result.TermTaken, result.CourseTakenDetails);

		if (result.Terms != null && result.Terms.length > 0) {
			details.enablePlan();
			details.setTerms(result.Terms);
		} else {
			details.disablePlan();
		}

		if (!allowAdd || result.HasExclusiveCourseInCoursePlanOrTranscript || !result.CanAddToPlan) {
			details.disableAddCourseButton();
			details.disableTermDropDown();
			//details.disablePlan();
		} else {
			details.enableAddCourseButton();
			details.enableTermDropDown();
		}
		
		details.showDetails();
	}
